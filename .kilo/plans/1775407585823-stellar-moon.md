## Implémentation

### Étape 1: Créer un script de test
- Créer `test_elisions_chapitre2.py` qui génère un fichier MP3 avec plusieurs phrases de test
- Chaque phrase sera séparée par une pause de 2 secondes
- Le SSML généré sera affiché pour vérification

### Étape 2: Exécuter et valider
- Lancer le script pour générer `output_audio/test_elisions_chapitre2.mp3`
- L'utilisateur écoute et valide les prononciations
- Ajuster le regex si nécessaire

## Script proposé

```python
#!/usr/bin/env python3
"""Test audio pour les élisions - Chapitre 2."""

import os
import html
import yaml
from google.cloud import texttospeech
from utils import check_google_credentials, fix_french_elisions_ssml

def load_config(config_path="config_google_tts.yaml"):
    if os.path.exists(config_path):
        with open(config_path, "r", encoding="utf-8") as f:
            return yaml.safe_load(f)
    return {}

def main():
    if not check_google_credentials():
        return

    config = load_config()
    voice_name = config.get("voice_name", "fr-FR-Chirp3-HD-Algenib")
    output_dir = config.get("output_dir", "output_audio")
    os.makedirs(output_dir, exist_ok=True)

    test_phrases = [
        "peu d'adultes possédaient",
        "n'était pas une demeure",
        "c'était un organisme vivant",
        "l'artefact reposait",
        "qu'il avait déterré",
        "s'insinuait dans sa moelle",
        "n'avait plus besoin",
        "j'ai juste l'habitude",
        "s'était accordé à lui",
    ]

    # Générer SSML avec pauses
    ssml_parts = []
    for phrase in test_phrases:
        safe_text = html.escape(phrase).replace("&#x27;", "'").replace("&quot;", '"')
        processed_text = fix_french_elisions_ssml(safe_text)
        ssml_parts.append(processed_text)
        print(f"Phrase: {phrase}")
        print(f"  SSML: {processed_text}")

    ssml_payload = f"<speak>{' <break time=\"2s\"/> '.join(ssml_parts)}</speak>"

    print(f"\nVoix: {voice_name}")
    print(f"Génération de l'audio...")

    client = texttospeech.TextToSpeechClient()
    synthesis_input = texttospeech.SynthesisInput(ssml=ssml_payload)
    voice = texttospeech.VoiceSelectionParams(language_code="fr-FR", name=voice_name)
    audio_config = texttospeech.AudioConfig(audio_encoding=texttospeech.AudioEncoding.MP3)

    response = client.synthesize_speech(input=synthesis_input, voice=voice, audio_config=audio_config)

    output_path = os.path.join(output_dir, "test_elisions_chapitre2.mp3")
    with open(output_path, "wb") as out:
        out.write(response.audio_content)

    print(f"Fichier: {output_path}")

if __name__ == "__main__":
    main()
```
